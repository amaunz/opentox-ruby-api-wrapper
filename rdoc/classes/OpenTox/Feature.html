                def initialize(params)
                        if params[:uri]
                                @uri = params[:uri].to_s
                        else
                                @uri = ENV['OPENTOX_FEATURES'] + uri_escape(params[:name]) 
                                params[:values].each do |k,v|
                                        @uri += '/' + k.to_s + '/' + v.to_s
                                end
                        end
